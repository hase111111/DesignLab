13namespace jsoncons { namespace msgpack {
14
15enum class msgpack_errc
16{
17 success = 0,
18 unexpected_eof = 1,
19 source_error,
20 invalid_utf8_text_string,
21 array_length_required,
22 object_length_required,
23 too_many_items,
24 too_few_items,
25 max_nesting_depth_exceeded,
26 length_is_negative,
27 invalid_timestamp,
28 unknown_type
29};
30
31class msgpack_error_category_impl
32 : public std::error_category
33{
34public:
35 const char* name() const noexcept override
36 {
37 return "jsoncons/msgpack";
38 }
39 std::string message(int ev) const override
40 {
41 switch (static_cast<msgpack_errc>(ev))
42 {
43 case msgpack_errc::unexpected_eof:
44 return "Unexpected end of file";
45 case msgpack_errc::source_error:
46 return "Source error";
47 case msgpack_errc::invalid_utf8_text_string:
48 return "Illegal UTF-8 encoding in text string";
49 case msgpack_errc::array_length_required:
50 return "MessagePack encoder requires array length";
51 case msgpack_errc::object_length_required:
52 return "MessagePack encoder requires object length";
53 case msgpack_errc::too_many_items:
54 return "Too many items were added to a MessagePack object or array";
55 case msgpack_errc::too_few_items:
56 return "Too few items were added to a MessagePack object or array";
57 case msgpack_errc::max_nesting_depth_exceeded:
58 return "Data item nesting exceeds limit in options";
59 case msgpack_errc::length_is_negative:
60 return "Request for the length of an array, map or string returned a negative result";
61 case msgpack_errc::invalid_timestamp:
62 return "Invalid timestamp";
63 case msgpack_errc::unknown_type:
64 return "An unknown type was found in the stream";
65 default:
66 return "Unknown MessagePack parser error";
67 }
68 }
69};
70
71inline
72const std::error_category& msgpack_error_category()
73{
74 static msgpack_error_category_impl instance;
75 return instance;
76}
77
78inline
79std::error_code make_error_code(msgpack_errc e)
80{
81 return std::error_code(static_cast<int>(e),msgpack_error_category());
82}
