67 void Calculate()
68 {
69 //データが空の場合は，平均値，分散，標準偏差をnulloptにする．
70 if (data_.empty())
71 {
72 average_ = std::nullopt;
73 variance_ = std::nullopt;
74 standard_deviation_ = std::nullopt;
75 return;
76 }
77
78 // 平均値を求める
79 average_ = sum_.value() / data_.size();
80
81 // 分散を求める
82 T variance = 0;
83 for (const auto& data : data_)
84 {
85 variance += (data - average_.value()) * (data - average_.value());
86 }
87 variance_ = variance / data_.size();
88
89 // 標準偏差を求める
90 standard_deviation_ = sqrt(variance_.value());
91 }
