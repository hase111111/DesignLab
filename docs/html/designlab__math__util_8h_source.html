80template <typename T>
81T ApproachTarget(const T& current, const T& target, float rate)
82{
83 assert(0 <= rate);
84 assert(rate <= 1); //0 <= rate <= 1 でなければならない
85
86 if (current == target) { return current; }
87
88 return static_cast<T>(current * (1 - rate) + target * rate);
89}
