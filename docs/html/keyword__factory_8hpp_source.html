45 template <class Json>
46 class keyword_factory : public subschema_validator_factory<Json>
47 {
48 public:
49 using reference_validator_type = reference_validator<Json>;
50 using validator_type = typename std::unique_ptr<keyword_validator<Json>>;
51 using validator_pointer = typename keyword_validator<Json>::self_pointer;
52 private:
53 struct subschema_registry
54 {
55 std::map<std::string, validator_pointer> schemas; // schemas
56 std::map<std::string, reference_schema<Json>*> unresolved; // unresolved references
57 std::map<std::string, Json> unprocessed_keywords;
58 };

86 validator_type make_subschema_validator(const Json& schema,
87 const compilation_context& context,
88 const std::vector<std::string>& keys) override
89 {
90 auto new_context = context.update_uris(schema, keys);
91
92 validator_pointer sch = nullptr;
93
94 switch (schema.type())
95 {
96 case json_type::bool_value:
97 if (schema.template as<bool>())
98 {
99 auto ref = true_validator<Json>::compile(new_context);
100 sch = ref.get();
101 subschemas_.emplace_back(std::move(ref));
102 }
103 else
104 {
105 auto ref = false_validator<Json>::compile(new_context);
106 sch = ref.get();
107 subschemas_.emplace_back(std::move(ref));
108 }
109 break;
110 case json_type::object_value:
111 {
112 auto it = schema.find("definitions");
113 if (it != schema.object_range().end())
114 {
115 for (const auto& def : it->value().object_range())
116 make_subschema_validator(def.value(), new_context, {"definitions", def.key()});
117 }
118
119 it = schema.find("$ref");
120 if (it != schema.object_range().end()) // this schema is a reference
121 {
122 schema_location relative(it->value().template as<std::string>());
123
124 auto id = new_context.resolve_back(relative);
125 //sch = get_or_create_reference(id);
126 auto ref = get_or_create_reference(id);
127 sch = ref.get();
128 subschemas_.emplace_back(std::move(ref));
129 }
130 else
131 {
132 auto ref = jsoncons::make_unique<type_validator<Json>>(this, schema, new_context);
133 sch = ref.get();
134 subschemas_.emplace_back(std::move(ref));
135 }
136 break;
137 }
138 default:
139 JSONCONS_THROW(schema_error("invalid JSON-type for a schema for " + new_context.uris()[0].string() + ", expected: boolean or object"));
140 break;
141 }
142
143 for (const auto& uri : new_context.uris())
144 {
145 insert_schema(uri, sch);
146
147 if (schema.type() == json_type::object_value)
148 {
149 for (const auto& item : schema.object_range())
150 insert_unknown_keyword(uri, item.key(), item.value()); // save unknown keywords for later reference
151 }
152 }
153 return jsoncons::make_unique<reference_validator_type>(sch);
154 }

156 void load_root(const Json& sch)
157 {
158 if (sch.is_object())
159 {
160 auto it = sch.find("$schema");
161 if (it != sch.object_range().end())
162 {
163 auto sv = it->value().as_string_view();
164 if (!schema_version::contains(sv))
165 {
166 std::string message("Unsupported schema version ");
167 message.append(sv.data(), sv.size());
168 JSONCONS_THROW(schema_error(message));
169 }
170 }
171 }
172 load(sch);
173 }
174
175 void load(const Json& sch)
176 {
177 subschema_registries_.clear();
178 root_ = make_subschema_validator(sch, compilation_context(schema_location("#")), {});
179
180 // load all external schemas that have not already been loaded
181
182 std::size_t loaded_count = 0;
183 do
184 {
185 loaded_count = 0;
186
187 std::vector<std::string> locations;
188 for (const auto& item : subschema_registries_)
189 locations.push_back(item.first);
190
191 for (const auto& loc : locations)
192 {
193 if (subschema_registries_[loc].schemas.empty()) // registry for this file is empty
194 {
195 if (resolver_)
196 {
197 Json external_schema = resolver_(loc);
198 make_subschema_validator(external_schema, compilation_context(schema_location(loc)), {});
199 ++loaded_count;
200 }
201 else
202 {
203 JSONCONS_THROW(schema_error("External schema reference '" + loc + "' needs to be loaded, but no resolver provided"));
204 }
205 }
206 }
207 }
208 while (loaded_count > 0);
209
210 for (const auto &file : subschema_registries_)
211 {
212 if (!file.second.unresolved.empty())
213 {
214 JSONCONS_THROW(schema_error("After all files have been parsed, '" +
215 (file.first == "" ? "<root>" : file.first) +
216 "' has still undefined references."));
217 }
218 }
219 }

223 void insert_schema(const schema_location& uri, validator_pointer s)
224 {
225 auto& file = get_or_create_file(std::string(uri.base()));
226 auto schemas_it = file.schemas.find(std::string(uri.fragment()));
227 if (schemas_it != file.schemas.end())
228 {
229 JSONCONS_THROW(schema_error("schema with " + uri.string() + " already inserted"));
230 return;
231 }
232
233 file.schemas.insert({std::string(uri.fragment()), s});
234
235 // is there an unresolved reference to this newly inserted schema?
236 auto unresolved_it = file.unresolved.find(std::string(uri.fragment()));
237 if (unresolved_it != file.unresolved.end())
238 {
239 unresolved_it->second->set_referred_schema(s);
240 file.unresolved.erase(unresolved_it);
241 }
242 }
243
244 void insert_unknown_keyword(const schema_location& uri,
245 const std::string& key,
246 const Json& value)
247 {
248 auto &file = get_or_create_file(std::string(uri.base()));
249 auto new_u = uri.append(key);
250 schema_location new_uri(new_u);
251
252 if (new_uri.has_fragment() && !new_uri.has_identifier())
253 {
254 auto fragment = std::string(new_uri.fragment());
255 // is there a reference looking for this unknown-keyword, which is thus no longer a unknown keyword but a schema
256 auto unresolved = file.unresolved.find(fragment);
257 if (unresolved != file.unresolved.end())
258 make_subschema_validator(value, compilation_context(new_uri), {});
259 else // no, nothing ref'd it, keep for later
260 file.unprocessed_keywords[fragment] = value;
261
262 // recursively add possible subschemas of unknown keywords
263 if (value.type() == json_type::object_value)
264 for (const auto& subsch : value.object_range())
265 {
266 insert_unknown_keyword(new_uri, subsch.key(), subsch.value());
267 }
268 }
269 }

271 validator_type get_or_create_reference(const schema_location& uri)
272 {
273 auto &file = get_or_create_file(std::string(uri.base()));
274
275 // a schema already exists
276 auto sch = file.schemas.find(std::string(uri.fragment()));
277 if (sch != file.schemas.end())
278 return jsoncons::make_unique<reference_validator_type>(sch->second);
279
280 // referencing an unknown keyword, turn it into schema
281 //
282 // an unknown keyword can only be referenced by a JSONPointer,
283 // not by a plain name identifier
284 if (uri.has_fragment() && !uri.has_identifier())
285 {
286 std::string fragment = std::string(uri.fragment());
287 auto unprocessed_keywords_it = file.unprocessed_keywords.find(fragment);
288 if (unprocessed_keywords_it != file.unprocessed_keywords.end())
289 {
290 auto &subsch = unprocessed_keywords_it->second;
291 auto s = make_subschema_validator(subsch, compilation_context(uri), {}); // A JSON Schema MUST be an object or a boolean.
292 file.unprocessed_keywords.erase(unprocessed_keywords_it);
293 return s;
294 }
295 }
296
297 // get or create a reference_schema
298 auto ref = file.unresolved.find(std::string(uri.fragment()));
299 if (ref != file.unresolved.end())
300 {
301 //return ref->second; // unresolved, use existing reference
302 return jsoncons::make_unique<reference_validator_type>(ref->second);
303 }
304 else
305 {
306 auto orig = jsoncons::make_unique<reference_schema<Json>>(uri.string());
307 auto p = file.unresolved.insert(ref,
308 {std::string(uri.fragment()), orig.get()})
309 ->second; // unresolved, create new reference
310
311 subschemas_.emplace_back(std::move(orig));
312 return jsoncons::make_unique<reference_validator_type>(p);
313 }
314 }
315
316 subschema_registry& get_or_create_file(const std::string& loc)
317 {
318 auto file = subschema_registries_.find(loc);
319 if (file != subschema_registries_.end())
320 return file->second;
321 else
322 return subschema_registries_.insert(file, {loc, {}})->second;
323 }

325 };
