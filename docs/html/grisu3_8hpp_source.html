48// diy_fp
49
50typedef struct diy_fp_t {
51 uint64_t f;
52 int e;
53} diy_fp_t;
54
55inline
56diy_fp_t minus(diy_fp_t x, diy_fp_t y)
57{
58 assert(x.e == y.e);
59 assert(x.f >= y.f);
60 diy_fp_t r = { x.f = x.f - y.f, x.e };
61 return r;
62}
63
64inline
65diy_fp_t multiply(diy_fp_t x, diy_fp_t y)
66{
67 uint64_t a, b, c, d, ac, bc, ad, bd, tmp;
68 diy_fp_t r; uint64_t M32 = 0xFFFFFFFF;
69 a = x.f >> 32; b = x.f & M32;
70 c = y.f >> 32; d = y.f & M32;
71 ac = a * c; bc = b * c; ad = a * d; bd = b * d;
72 tmp = (bd >> 32) + (ad & M32) + (bc & M32);
73 tmp += 1U << 31;
74 r.f = ac + (ad >> 32) + (bc >> 32) + (tmp >> 32);
75 r.e = x.e + y.e + 64;
76 return r;
77}

79// k_comp
80
81inline
82int k_comp(int e, int alpha, int /*gamma*/)
83{
84 constexpr double d_1_log2_10 = 0.30102999566398114; // 1 / lg(10)
85 int x = alpha - e + 63;
86 return static_cast<int>(std::ceil(x * d_1_log2_10));
87}

126constexpr int dp_significand_size = 52;
127constexpr int dp_exponent_bias = (0x3FF + dp_significand_size);
128constexpr int dp_min_exponent = (-dp_exponent_bias);
129constexpr uint64_t dp_exponent_mask = 0x7FF0000000000000;
130constexpr uint64_t dp_significand_mask = 0x000FFFFFFFFFFFFF;
131constexpr uint64_t dp_hidden_bit = 0x0010000000000000;

133inline
134diy_fp_t normalize_diy_fp(diy_fp_t in)
135{
136 diy_fp_t res = in;
137 /* Normalize now */
138 /* the original number could have been a denormal. */
139 while (!(res.f & dp_hidden_bit))
140 {
141 res.f <<= 1;
142 res.e--;
143 }
144 /* do the final shifts in one go. Don't forget the hidden bit (the '-1') */
145 res.f <<= (diy_significand_size - dp_significand_size - 1);
146 res.e = res.e - (diy_significand_size - dp_significand_size - 1);
147 return res;
148}

150inline
151diy_fp_t double2diy_fp(double d)
152{
153 uint64_t d64 = double_to_uint64(d);
154 int biased_e = (d64 & dp_exponent_mask) >> dp_significand_size;
155 uint64_t significand = (d64 & dp_significand_mask);
156 diy_fp_t res;
157 if (biased_e != 0)
158 {
159 res.f = significand + dp_hidden_bit;
160 res.e = biased_e - dp_exponent_bias;
161 }
162 else
163 {
164 res.f = significand;
165 res.e = dp_min_exponent + 1;
166 }
167 return res;
168}

172{
173 diy_fp_t res = in;
174 /* Normalize now */
175 /* the original number could have been a denormal. */
176 while (!(res.f & (dp_hidden_bit << 1)))
177 {
178 res.f <<= 1;
179 res.e--;
180 }
181 /* do the final shifts in one go. Don't forget the hidden bit (the '-1') */
182 res.f <<= (diy_significand_size - dp_significand_size - 2);
183 res.e = res.e - (diy_significand_size - dp_significand_size - 2);
184 return res;
185}

187inline
188void normalized_boundaries(double d, diy_fp_t *out_m_minus, diy_fp_t *out_m_plus)
189{
190 diy_fp_t v = double2diy_fp(d);
191 diy_fp_t pl, mi;
192 bool significand_is_zero = v.f == dp_hidden_bit;
193 pl.f = (v.f << 1) + 1; pl.e = v.e - 1;
194 pl = normalize_boundary(pl);
195 if (significand_is_zero)
196 {
197 mi.f = (v.f << 2) - 1;
198 mi.e = v.e - 2;
199 } else
200 {
201 mi.f = (v.f << 1) - 1;
202 mi.e = v.e - 1;
203 }
204 int x = mi.e - pl.e;
205 mi.f <<= x;
206 mi.e = pl.e;
207 *out_m_plus = pl;
208 *out_m_minus = mi;
209}

226inline
227bool round_weed(char *buffer, int len,
228 uint64_t wp_W, uint64_t Delta,
229 uint64_t rest, uint64_t ten_kappa,
230 uint64_t ulp)
231{
232 uint64_t wp_Wup = wp_W - ulp;
233 uint64_t wp_Wdown = wp_W + ulp;
234 while (rest < wp_Wup &&
235 Delta - rest >= ten_kappa &&
236 (rest + ten_kappa < wp_Wup ||
237 wp_Wup - rest >= rest + ten_kappa - wp_Wup))
238 {
239 buffer[len - 1]--; rest += ten_kappa;
240 }
241 if (rest < wp_Wdown &&
242 Delta - rest >= ten_kappa &&
243 (rest + ten_kappa < wp_Wdown ||
244 wp_Wdown - rest > rest + ten_kappa - wp_Wdown)) return 0;
245 return 2 * ulp <= rest && rest <= Delta - 4 * ulp;
246}

248inline
249bool digit_gen(diy_fp_t Wm, diy_fp_t W, diy_fp_t Wp,
250 char *buffer, int *len, int *K)
251{
252 const uint32_t TEN2 = 100;
253
254 uint32_t div, p1; uint64_t p2, tmp, unit = 1;
255 int d, kappa;
256 diy_fp_t one, wp_W, Delta;
257 Delta = minus(Wp, Wm); Delta.f += 2 * unit;
258 wp_W = minus(Wp, W); wp_W.f += unit;
259 one.f = ((uint64_t)1) << -Wp.e; one.e = Wp.e;
260 p1 = static_cast<uint32_t>((Wp.f + 1) >> -one.e);
261 p2 = (Wp.f + 1) & (one.f - 1);
262 *len = 0; kappa = 3; div = TEN2;
263 while (kappa > 0)
264 {
265 d = p1 / div;
266 if (d || *len) buffer[(*len)++] = (char)('0' + d);
267 p1 %= div; kappa--;
268 tmp = (((uint64_t)p1) << -one.e) + p2;
269 if (tmp < Delta.f)
270 {
271 *K += kappa;
272 return round_weed(buffer, *len, wp_W.f, Delta.f, tmp,
273 ((uint64_t)div) << -one.e, unit);
274 }
275 div /= 10;
276 }
277 while (1)
278 {
279 p2 *= 10; Delta.f *= 10; unit *= 10;
280 d = static_cast<int>(p2 >> -one.e);
281 if (d || *len) buffer[(*len)++] = (char)('0' + d);
282 p2 &= one.f - 1; kappa--;
283 if (p2 < Delta.f)
284 {
285 *K += kappa;
286 return round_weed(buffer, *len, wp_W.f * unit, Delta.f, p2,
287 one.f, unit);
288 }
289 }

292inline
293bool grisu3(double v, char *buffer, int *length, int *K)
294{
295 diy_fp_t w_m, w_p;
296 int q = 64, alpha = -59, gamma = -56;
297 normalized_boundaries(v, &w_m, &w_p);
298 diy_fp_t w = normalize_diy_fp(double2diy_fp(v));
299 int mk = k_comp(w_p.e + q, alpha, gamma);
300 diy_fp_t c_mk = cached_power(mk);
301 diy_fp_t W = multiply(w, c_mk);
302 diy_fp_t Wp = multiply(w_p, c_mk);
303 diy_fp_t Wm = multiply(w_m, c_mk);
304 *K = -mk;
305 bool result = digit_gen(Wm, W, Wp, buffer, length, K);
306 buffer[*length] = 0;
307 return result;
308}
