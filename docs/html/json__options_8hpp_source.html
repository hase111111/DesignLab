50struct default_json_parsing
51{
52 bool operator()(json_errc ec, const ser_context&) noexcept
53 {

67{
68 bool operator()(const std::error_code& ec, const ser_context&) noexcept
69 {
70 return ec == json_errc::illegal_comment || ec == jsoncons::json_errc::extra_comma;

243 return neg_inf_replacement_;

249 }

259 else if (can_read_nan_replacement_ && nan_replacement_.size() >= 2) // string

263#endif

281#endif

851private:
852 enum class input_state {initial,begin_quote,character,end_quote,escape,error};
853 bool is_string(const string_type& s) const
854 {
855 input_state state = input_state::initial;
856 for (char_type c : s)
857 {
858 switch (c)
859 {
860 case '\t': case ' ': case '\n': case'\r':
861 break;
862 case '\\':
863 state = input_state::escape;
864 break;
865 case '\"':
866 switch (state)
867 {
868 case input_state::initial:
869 state = input_state::begin_quote;
870 break;
871 case input_state::begin_quote:
872 case input_state::character:
873 state = input_state::end_quote;
874 break;
875 case input_state::end_quote:
876 state = input_state::error;
877 break;
878 case input_state::escape:
879 state = input_state::character;
880 break;
881 default:
882 state = input_state::character;
883 break;
884 }
885 break;
886 default:
887 break;
888 }
889
890 }
891 return state == input_state::end_quote;
892 }
