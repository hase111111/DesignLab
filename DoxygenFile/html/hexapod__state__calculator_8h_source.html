123 dl_vec::SVector getLocalCoxaJointPos(const int leg_num) const
124 {
125 if (leg_num == 0) { return dl_vec::SVector(HexapodConst::BODY_FRONT_LENGTH, -HexapodConst::BODY_FRONT_WIDTH, 0.0f); } // 脚0 右上

130 else if (leg_num == 5) { return dl_vec::SVector(HexapodConst::BODY_FRONT_LENGTH, HexapodConst::BODY_FRONT_WIDTH, 0.0f); } // 脚5 左上
