76void myDxlib3DFunc::drawHexagonalPrism(const VECTOR _vertex[6], const float _height, const unsigned int _color)
77{
78 //6角形面の法線方向のベクトルを取得する．やっている処理としては，頂点0から1へ行くベクトルをv01，同様に頂点0から2へ行くベクトルをv02とすると，
79 // v01とv02の外積(Cross)をとると法線方向のベクトルが取得できるため，これを単位ベクトルに変換(Norm，ノーマライズのこと)し，高さの半分だけ倍にした．
80 const VECTOR _center_to_top = VScale(VNorm(VCross(VSub(_vertex[0], _vertex[1]), VSub(_vertex[0], _vertex[2]))), _height / 2.0f);
81
82 //上面の頂点．
83 const VECTOR _vertex_top[6] =
84 {
85 VAdd(_vertex[0],_center_to_top),
86 VAdd(_vertex[1],_center_to_top),
87 VAdd(_vertex[2],_center_to_top),
88 VAdd(_vertex[3],_center_to_top),
89 VAdd(_vertex[4],_center_to_top),
90 VAdd(_vertex[5],_center_to_top)
91 };
92
93 //底面の頂点
94 const VECTOR _vertex_bottom[6] =
95 {
96 VSub(_vertex[0],_center_to_top),
97 VSub(_vertex[1],_center_to_top),
98 VSub(_vertex[2],_center_to_top),
99 VSub(_vertex[3],_center_to_top),
100 VSub(_vertex[4],_center_to_top),
101 VSub(_vertex[5],_center_to_top)
102 };
103
104 drawHexagon(_vertex_top, _color); //上面を描画する．
105 drawHexagon(_vertex_bottom, _color); //底面を描画する．
106
107 //側面を描画していく．側面は四角形6つで構成されるので，3角形が12こ必要になる．
108 for (int i = 0; i < 6; i++)
109 {
110 DrawTriangle3D(_vertex_top[i % 6], _vertex_top[(i + 1) % 6], _vertex_bottom[i % 6], _color, TRUE);
111 DrawTriangle3D(_vertex_top[(i + 1) % 6], _vertex_bottom[i % 6], _vertex_bottom[(i + 1) % 6], _color, TRUE);
112 }
113
114}
