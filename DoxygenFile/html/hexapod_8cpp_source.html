20//ベクトルの3Dローテーション オイラー角（テイト・ブライアン角）Y-X-Z
21// 2018.12.27 コメントのY-X-Zって間違いっぽい．英語版wikipediaに詳しいが，多分Y-X-Yあたりのオイラー角の定義だと思うんですが……（テイト・ブライアンも違うのでは）
22//2020.05.15.hato
23//　これは、RPYが基本的に胴体の傾き（別に胴体じゃなくてもいいけど）だから、胴体座標系やcoxa座標系から座標ΣP（例えばcoxaからみた脚先座標など）を
24//グローバル座標で見た座標Σ0Pにへんかんするための、
25// Σ0_P = （Σ0_R_Σ)ΣP + (グローバル座標系で見たその座標系の原点座標）
26//っていう式の（Σ0_R_Σ)ΣPの計算をしてる関数。
27//
28//（Σ0RΣ)は回転行列で（Σ0RΣ)=Ry*Rx*Rz
29//Ry,Rx,Rzは添え字の軸回りに回転させたときの回転行列をあらわす。
30//一般的には、進行方向にｘ軸をとるから、
31//Rxがロールの関数＝Rx(thR), Ryがピッチの関数=Ry(thP),Pz がヨーの関数=Rz(thY)
32//になる。
33//ただ、このシミュレーションでは、進行方向がy軸だから、
34//Rxがピッチの関数＝Rx(thR), Ryがロールの関数=Ry(thP),Pz がヨーの関数=Rz(thY)
35//になっている。
36//また、Σ0RΣ　＝　R＊P＊Yの順に回転行列を計算するのが一般的？だから、それにあわせると、
37//普通は、Σ0RΣ　＝　Rx(R) * Ry(P) * Rz(Y)　の順番で計算するところ
38//ここでは、Σ0RΣ　＝　Ry(R) * Rx(P) * Rz(Y)　の順番で計算している。
39//そこがちょっとややこしいところだけど、計算自体は間違ってない。
40my_vec::SVector Hexapod::rotation(const my_vec::SVector& In, const my_vec::SVector& center, const float thP, const float thR, const float thY) const
41{
42 my_vec::SVector ans, buf;
43 buf = In - center;
44 ans.x = (cos(thR) * cos(thY) + sin(thR) * sin(thP) * sin(thY)) * buf.x +
45 (cos(thY) * sin(thR) * sin(thP) - cos(thR) * sin(thY)) * buf.y +
46 (cos(thP) * sin(thR)) * buf.z;
47
48 ans.y = cos(thP) * sin(thY) * buf.x +
49 cos(thP) * cos(thY) * buf.y +
50 (-sin(thP)) * buf.z;
51
52 ans.z = (cos(thR) * sin(thP) * sin(thY) - cos(thY) * sin(thR)) * buf.x +
53 (cos(thR) * cos(thY) * sin(thP) + sin(thR) * sin(thY)) * buf.y +
54 (cos(thR) * cos(thP)) * buf.z;
55
56 return ans + center;
57}

59bool Hexapod::setJointPosi()
60{
61 // 逆運動学の計算結果を用いて準運動学を計算する．逆運動学，運動学の計算から算術する方式 ロボット固有
62 using namespace my_math;
63
64 //逆運動学と運動学を行った結果が半径Permission^0.5の円の中なら等しいと考える
65 const float PERMISSION = 1;
66
67 const float mins[3] = { -1.428f, -1.780f, -1.194f }; //脚可動範囲
68 const float maxs[3] = { 1.402f, 1.744f, 1.769f }; //左からcoxa,femur,tibia
69 const float coxaMins[6] = { -0.610f, -1.428f, -2.213f, -0.960f, -1.745f, -2.531f };
70 const float coxaMaxs[6] = { 2.187f, 1.402f, 0.617f, 2.531f, 1.745f, 0.960f };
71
72 for (int legNum = 0; legNum < HexapodConst::LEG_NUM; legNum++)
73 {
74 const my_vec::SVector legposi(m_local_leg_pos[legNum].x, m_local_leg_pos[legNum].y, -m_local_leg_pos[legNum].z);
75
76 //逆運動学
77 // first, make this a 2DOF problem... by solving coxa
78 const float coxa = atan2(legposi.y, legposi.x);//coxa角度
79
80 const float IK_trueX = sqrt(squared(legposi.x) + squared(legposi.y)) - HexapodConst::COXA_LENGTH;//xy平面におけるfemur脚先までの距離
81 const float im = sqrt(squared(IK_trueX) + squared(legposi.z)); //femurから脚先までの距離
82
83 // get femur angle above horizon...
84 float q1 = -atan2(legposi.z, IK_trueX);//femurから脚先へのベクトルとxy平面との角度
85 float d1 = squared(HexapodConst::FEMUR_LENGTH) - squared(HexapodConst::TIBIA_LENGTH) + squared(im);
86 float d2 = 2.0f * HexapodConst::FEMUR_LENGTH * im;
87 float q2 = acos(d1 / d2);
88 const float femur = q1 + q2;
89
90 // and tibia angle from femur...
91 d1 = pow(HexapodConst::FEMUR_LENGTH, 2.0f) - squared(im) + squared(HexapodConst::TIBIA_LENGTH);
92 d2 = 2.0f * HexapodConst::TIBIA_LENGTH * HexapodConst::FEMUR_LENGTH;
93 const float tibia = acos(d1 / d2) - my_math::MY_FLT_PI / 2.0f;
94
95 //運動学
96 const float K_trueX = HexapodConst::FEMUR_LENGTH * cos(femur) + HexapodConst::TIBIA_LENGTH * cos(femur + tibia - my_math::MY_FLT_PI / 2.0f);
97
98 my_vec::SVector kinematics; //ans of kinematics use sorution of i_kinematics
99 kinematics.x = cos(coxa) * (K_trueX + HexapodConst::COXA_LENGTH);
100 kinematics.y = sin(coxa) * (K_trueX + HexapodConst::COXA_LENGTH);
101 kinematics.z = -(HexapodConst::FEMUR_LENGTH * sin(femur) + HexapodConst::TIBIA_LENGTH * sin(femur + tibia - my_math::MY_FLT_PI / 2.0f));
102
103 m_local_femurjoint_pos[legNum] = m_local_coxajoint_pos[legNum] + my_vec::SVector(HexapodConst::COXA_LENGTH * cos(coxa), HexapodConst::COXA_LENGTH * sin(coxa), 0.0f);
104 m_local_tibiajoint_pos[legNum] = m_local_femurjoint_pos[legNum] + my_vec::SVector(HexapodConst::FEMUR_LENGTH * cos(femur) * cos(coxa), HexapodConst::FEMUR_LENGTH * cos(femur) * sin(coxa), HexapodConst::FEMUR_LENGTH * sin(femur));
105
106 const float Permission = (kinematics - legposi).lengthSquare();
107
108 if (PERMISSION < Permission) { std::cout << "Error PERMISSION OVER \n"; return true; };
109 }
110
111 return false;

228
229 //i番目の脚先からロボットの重心へ向かうベクトル．

231 i_to_com.z = 0.0f;
232
233 //脚がクロスする場合
234 if (i_to_i_puls_1.cross(i_to_com).z > 0.0f)
235 {
236 return false;
237 }
238 }
239

337 if ((t1 < -2.0f * my_math::MY_FLT_PI) || (t1 > 2.0f * my_math::MY_FLT_PI)) { t1 = fmod(t1, 2.0f * my_math::MY_FLT_PI); } //-2pi~2pに
338
339 if (t1 < -my_math::MY_FLT_PI) { t1 = t1 + 2.0f * my_math::MY_FLT_PI; } //-piより小さければ
340
341 if (t2 > my_math::MY_FLT_PI) { t2 = t2 - 2.0f * my_math::MY_FLT_PI; }
342
343 //初期化
344 p1 = my_vec::SVector(0.0f, 0.0f, 0.0f);
345 p2 = my_vec::SVector(0.0f, 0.0f, 0.0f);
346
347 //p1p2の座標決定
348 if ((r * cos(t1)) > p2.x)
349 {
350 p2.x = r * cos(t1);
351 }
352 else if ((r * cos(t1)) < p1.x)
353 {
354 p1.x = r * cos(t1);
355 }
356
357 if ((r * sin(t1)) > p2.y)
358 {
359 p2.y = r * sin(t1);
360 }
361 else if ((r * sin(t1)) < p1.y)
362 {
363 p1.y = r * sin(t1);
364 }
365
366 if ((r * cos(t2)) > p2.x)
367 {
368 p2.x = r * cos(t2);
369 }
370 else if ((r * cos(t2)) < p1.x)
371 {
372 p1.x = r * cos(t2);
373 }
374
375 if ((r * sin(t2)) > p2.y)
376 {
377 p2.y = r * sin(t2);
378 }
379 else if ((r * sin(t2)) < p1.y)
380 {
381 p1.y = r * sin(t2);
382 }
383
384 //90 180 -90 -180度の時の処理
385 if ((t2 > my_math::MY_FLT_PI / 4.0f && t1 < my_math::MY_FLT_PI / 4.0f) || (t1 > 0.0f && t1 < my_math::MY_FLT_PI / 4.0f && t2 < -my_math::MY_FLT_PI / 4.0f)) { p2.y = r; }//r*sin(90)
386 if (t2 > 0.0f && t1 < 0.0f) { p2.x = r; } //r*con(0)
387 if ((t2 > -my_math::MY_FLT_PI / 4.0f && t1 < -my_math::MY_FLT_PI / 4.0f) || (t1 > my_math::MY_FLT_PI / 4.0f && t2 < 0.0f && t2 > -my_math::MY_FLT_PI / 4.0f)) { p1.y = -r; }//r*sin(-90)
388 if (t2 < t1) { p1.x = -r; } //r*cos(180)
389
390 //グローバル座標に変換
391 p1 = p1 + getGlobalCoxaJointPos(_legnum);
392 p2 = p2 + getGlobalCoxaJointPos(_legnum);
393}
