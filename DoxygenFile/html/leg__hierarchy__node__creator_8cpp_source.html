27void LegHierarchyNodeCreator::create(const SNode& current_node, const int current_node_index, std::vector<SNode>* output_graph)
28{
29 //現在，接地している脚の本数を数える
30 const int kLiftedLegNum = dl_leg::getLiftedLegNum(current_node.leg_state);
31
32 //遊脚している脚の本数によって処理をする
33 if (kLiftedLegNum == 1)
34 {
35 // 1 本遊脚している．
36 create1LegLifted(current_node, current_node_index, output_graph);
37 }
38 else if (kLiftedLegNum == 2)

40 // 2 本遊脚している．
41 create2LegLifted(current_node, current_node_index, output_graph);

44 {
45 // 3 本遊脚している．
46 create3LegLifted(current_node, current_node_index, output_graph);
47 }
48 else
49 {
50 //ここに来るのは接地している脚の数が6本 or 1本 or 2本．地面についている脚が3本を切ることはない，何故ならロボットが倒れてしまうため．
51 //
52 //また6本接地しているならば脚を動かせない(遊脚する必要がある)．よって処理を行わない．(そのままの状態を次のノードにする．)
53 SNode new_node = current_node;
54
55 new_node.changeNextNode(current_node_index, m_next_move); //次のノード用に，深さ・親・次の動作を更新する．

63void LegHierarchyNodeCreator::create1LegLifted(const SNode& current_node, const int current_node_index, std::vector<SNode>* output_graph)
64{
65 //遊脚している脚を探す．遊脚数は1なので1つの数字が帰るはず
66 std::vector<int> lifted_leg_list;
67 dl_leg::getLiftedLegNumWithVector(current_node.leg_state, lifted_leg_list);
68
69 //脚状態 0001(1) から 0111(7)まで 次のパターンを生成する．なお上位bitは遊脚を表す．(0なら遊脚)
70 for (int i = 1; i <= dl_leg::DISCRETE_NUM; ++i)
71 {
72 SNode new_node = current_node; //新しい脚状態を生成する.
73

75
