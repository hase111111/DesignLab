56void ComSelecter::makeComCandidatePoint(const my_vec::SPolygon2& polygon, std::vector<my_vec::SVector2>& coms) const
57{
58 //波東さんの処理では多角形を囲むような四角形を作るので，まずはそれを作る
59 const float kMinX = polygon.getMinX();
60 const float kMaxX = polygon.getMaxX();
61 const float kMinY = polygon.getMinY();
62 const float kMaxY = polygon.getMaxY();
63
64 const float kWidth = kMaxX - kMinX;
65 const float kHeight = kMaxY - kMinY;
66
67 const float kDeltaWidth = kWidth / (float)DISCRETIZATION_NUM;
68 const float kDeltaHeight = kHeight / (float)DISCRETIZATION_NUM;
69
70 //上記の四角形の中にある点を全て候補に追加する．ただし，多角形の中にある点のみ．
71 for (int x = 0; x <= DISCRETIZATION_NUM; ++x)
72 {
73 for (int y = 0; y <= DISCRETIZATION_NUM; ++y)
74 {
75 const my_vec::SVector2 kComPoint{kMinX + kDeltaWidth * x, kMinY + kDeltaHeight * y};
76
77 if (polygon.isInside(kComPoint))
78 {
79 coms.push_back(kComPoint);
80 }
81 }
82 }
83
84 if (DO_DEBUG_PRINT) { std::cout << "CandidatePointNum is " << coms.size() << std::endl; }
85}
