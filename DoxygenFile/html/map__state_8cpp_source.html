74void MapState::makeDevideMap()
75{
76 //マップを切り分ける四角形の辺の長さを算出する．
77 const float x_length = (MapConst::MAP_MAX_FORWARD - MapConst::MAP_MIN_FORWARD) / static_cast<float>(MapConst::LP_DIVIDE_NUM);
78 const float y_length = (MapConst::MAP_MAX_HORIZONTAL - MapConst::MAP_MIN_HORIZONTAL) / static_cast<float>(MapConst::LP_DIVIDE_NUM);
79
80 m_devide_map.resize(MapConst::LP_DIVIDE_NUM * MapConst::LP_DIVIDE_NUM);
81
82 //マップのデータ全てに処理をする．
83 for (const auto& i : m_map_data)
84 {
85 //xy方向のブロック番号をそれぞれ求める
86 int x = (int)((i.x - (float)MapConst::MAP_MIN_FORWARD) / x_length);
87 int y = (int)((i.y - (float)MapConst::MAP_MIN_HORIZONTAL) / y_length);
88
89 //マップの範囲内にいる時のみ追加する
90 if (0 <= x && x < MapConst::LP_DIVIDE_NUM)
91 {
92 if (0 <= y && y < MapConst::LP_DIVIDE_NUM)
93 {
94 //値を挿入する
95 m_devide_map[getDevideMapNum(x, y)].push_back(i);
96 }
97 }
98 }
99
100 // m_devide_map_top_zを更新する．
101 for (int i = 0; i < MapConst::LP_DIVIDE_NUM * MapConst::LP_DIVIDE_NUM; i++)
102 {
103 m_devide_map_top_z.push_back(kMapMinZ); //初期値は最小値．
104
105 //全ての要素を参照する．
106 for (const auto& i : m_devide_map[i])
107 {
108 //現在値と比べて大きいものを追加．
109 m_devide_map_top_z.back() = std::max(i.z, m_devide_map_top_z.back());
110 }
111 }
