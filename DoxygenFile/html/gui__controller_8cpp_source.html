31void GUIController::update(CameraController& camera_controller, const int max_node, int& display_node, const int counter)
32{
33 if (m_mode == ENodeDisplayNode::AUTO_UPDATE)
34 {
35 if (counter % CHANGE_NEXT_NODE == 0)
36 {
37 display_node++;
38 }
39 }
40 else if (m_mode == ENodeDisplayNode::ALWAYS_NEW)
41 {
42 display_node = max_node - 1;
43 }
44
45
46 //表示切替キーが押されたならば，表示切り返えフラグの値をさかさまにする．
47 if (Keyboard::getIns()->getPressingCount(KEY_INPUT_Z) == 1) { m_is_displayed = !m_is_displayed; }
48
49 if (m_is_displayed)
50 {
51 //表示していない時にボタンの処理を行わない
52
53 //ボタンを更新する．
54 for (int i = 0; i < (int)m_buttom.size(); i++)
55 {
56 m_buttom.at(i)->update();
57
58 //ボタンの追加された順番ごとに処理をする．クソみたいな実装でスマソ
59 switch (i)
60 {
61 case 0:
62 if (m_buttom.at(i)->isPushedNow()) { camera_controller.setCameraMode(ECameraMode::TopView); }
63 break;
64
65 case 1:
66 if (m_buttom.at(i)->isPushedNow()) { camera_controller.setCameraMode(ECameraMode::OverheadView); }
67 break;
68
69 case 2:
70 if (m_buttom.at(i)->isPushedNow()) { camera_controller.setCameraMode(ECameraMode::SideView); }
71 break;
72
73 case 3:
74 if (m_buttom.at(i)->isPushedNow()) { camera_controller.setCameraMode(ECameraMode::OverheadViewFlip); }
75 break;
76
77 case 4:
78 if (m_buttom.at(i)->isPushedNow()) { camera_controller.setCameraMode(ECameraMode::SideViewFlip); }
79 break;
80
81 case 5:
82 if (m_buttom.at(i)->isPushedNow())
83 {
84 if (m_mode == ENodeDisplayNode::SELECTABLE)m_mode = ENodeDisplayNode::ALWAYS_NEW;
85 else if (m_mode == ENodeDisplayNode::ALWAYS_NEW)m_mode = ENodeDisplayNode::AUTO_UPDATE;
86 else if (m_mode == ENodeDisplayNode::AUTO_UPDATE)m_mode = ENodeDisplayNode::SELECTABLE;
87 }
88 break;
89
90 case 6:
91 if (m_buttom.at(i)->isPushedNow()) { display_node--; }
92 break;
93
94 case 7:
95 if (m_buttom.at(i)->isPushedNow()) { display_node++; }
96 break;
97
98 default:
99 break;
100 }
101 }
102 }
103
104
105 if (display_node < 0) { display_node = 0; }
106 if (display_node >= max_node) { display_node = max_node - 1; }
107}
