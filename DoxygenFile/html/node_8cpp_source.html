94void SNode::init(const bool do_random)
95{
96 using my_vec::SVector;
97 using my_vec::SRotator;
98
99 //脚状態
100 const bool kLegGround[HexapodConst::LEG_NUM] = { true, true, true, true, true, true };
101 const int kLegPos[HexapodConst::LEG_NUM] = { 4, 4, 4, 4, 4, 4 };
102 leg_state = dl_leg::makeLegState(ComType::EComPattern::CENTER_BACK, kLegGround, kLegPos);
103
104 //脚付け根を原点とした，脚先の位置を初期化する．
105 const float kComZ = HexapodConst::VERTICAL_MIN_RANGE + MapConst::MAX_Z_BASE; // ロボットの重心のZ座標
106
107 for (int i = 0; i < HexapodConst::LEG_NUM; i++)
108 {
109 leg_pos[i] = leg_base_pos[i] = HexapodStateCalclator::getLocalBaseLegPos(i, -kComZ);
110 }
111
112 //グローバル座標の重心位置．グローバル座標(0,0,0)を中心とした，下の変数 _x，_yを半径とする楕円形のなかに重心を移動する．
113 const float kAngle = do_random ? my_math::generateRandomNumber(0.0f, 2.0f * my_math::MY_FLT_PI) : 0;
114 const float kEx = do_random ? my_math::generateRandomNumber(0.0f, 1.0f) : 0;
115
116 const float kX = ((float)MapConst::MAP_START_ROUGH - MapConst::MAP_MIN_FORWARD) * 0.25f;
117 const float kY = ((float)MapConst::MAP_MAX_HORIZONTAL - MapConst::MAP_MIN_HORIZONTAL) / 2.0f * 0.8f;
118
119 global_center_of_mass = my_vec::SVector(kEx * kX * cos(kAngle), kEx * kY * sin(kAngle), kComZ);
120
121 //ロールピッチヨーで回転を表現する．ロボットの重心を中心にして回転する．
122 rot = my_vec::SRotator(0, 0, 0);
123
124 next_move = EHexapodMove::COM_UP_DOWN;
125 parent_num = -1;
126 depth = 0;
127}

130void SNode::changeGlobalCenterOfMass(const my_vec::SVector& new_com, const bool base)
131{
132 const my_vec::SVector kComDif = new_com - global_center_of_mass;
133
134 global_center_of_mass = new_com;
135
136 for (int i = 0; i < HexapodConst::LEG_NUM; i++)
137 {
138 if (dl_leg::isGrounded(leg_state, i) == true)
139 {
140 leg_pos[i] -= kComDif;
141 leg_base_pos[i] = leg_pos[i];
142 }
143 else
144 {
145 if (base)leg_base_pos[i] = HexapodStateCalclator::getLocalBaseLegPos(i, leg_base_pos[i].z - kComDif.z);
146 }
147 }
