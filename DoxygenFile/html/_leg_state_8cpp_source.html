4int leg_state::makeLegState(const int _com_pattern, const bool _ground[HexapodConst::LEG_NUM], const int _leg_pos[HexapodConst::LEG_NUM])
5{
6 int res = 0;
7
8 if (0 <= _com_pattern && _com_pattern < ComType::COM_PATTERN_NUM)
9 {
10 res |= _com_pattern << SHIFT_TO_COM_NUM; //重心パターンの数値だけbitを立てる
11 }
12
13 for (int i = 0; i < HexapodConst::LEG_NUM; i++)
14 {
15 if (_ground[i] == true) { res |= 0b1000 << (i * 4); } //接地しているならば上位bitを立てる

18 {

20 res |= _leg_pos[i] << (i * 4);
21 }
22 else
23 {
24 //範囲外ならば 4 (基準位置)にする．
25 res |= 4 << (i * 4);
26 }
27 }
