45template <typename T>
46std::string EnumToStringRemoveTopK(const T& enum_value)
47{
48 //型チェック Tがenum型であることをチェックする．
49 static_assert(std::is_enum<T>::value, "引数はenum，あるいはenum classである必要があります．");
50
51 std::string str = static_cast<std::string>(magic_enum::enum_name(enum_value));
52
53 if (str.size() > 0 && str[0] == 'k')
54 {
55 // 先頭のkを削除する．
56 str.erase(0, 1);
57 }
58
59 return str;
60}

97template <typename T, typename = std::enable_if_t<std::is_enum<T>::value>>
98std::string EnumEntriesToString(const std::string separator)
99{
100 std::string str;
101 using enum_type = typename std::underlying_type<T>::type;
102 bool is_first = true;

112 is_first = false;

114

116 str += " = ";
117 str += std::to_string(static_cast<enum_type>(e));

124template <typename T>
125std::string GetTypeName(const T& type)
126{
127 std::string str = typeid(type).name();
128
129 std::vector<std::string> eliminate
130 {
131 "class ", "struct ", "designlab::", ",void"
132 };
133
134 for (const auto& e : eliminate)
135 {
136 while (true)
137 {
138 auto pos = str.find(e);
139
140 if (pos == std::string::npos)
141 {
142 break;
143 }
144
145 str.erase(pos, e.size());
146
147 }
148 }
149 return str;
