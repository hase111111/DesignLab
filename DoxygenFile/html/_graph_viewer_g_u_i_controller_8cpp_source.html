170void GraphViewerGUIController::inputNumber()
171{
172 // Cキーでリセット
173 if (Keyboard::getIns()->getPressingCount(KEY_INPUT_C) == 1)
174 {
175 m_input_number = -1;
176 return;
177 }
178
179 // 数字入力
180 int _temp = -1;
181
182 if (Keyboard::getIns()->getPressingCount(KEY_INPUT_0) == 1) { _temp = 0; }
183 else if (Keyboard::getIns()->getPressingCount(KEY_INPUT_1) == 1) { _temp = 1; }
184 else if (Keyboard::getIns()->getPressingCount(KEY_INPUT_2) == 1) { _temp = 2; }
185 else if (Keyboard::getIns()->getPressingCount(KEY_INPUT_3) == 1) { _temp = 3; }
186 else if (Keyboard::getIns()->getPressingCount(KEY_INPUT_4) == 1) { _temp = 4; }
187 else if (Keyboard::getIns()->getPressingCount(KEY_INPUT_5) == 1) { _temp = 5; }
188 else if (Keyboard::getIns()->getPressingCount(KEY_INPUT_6) == 1) { _temp = 6; }
189 else if (Keyboard::getIns()->getPressingCount(KEY_INPUT_7) == 1) { _temp = 7; }
190 else if (Keyboard::getIns()->getPressingCount(KEY_INPUT_8) == 1) { _temp = 8; }
191 else if (Keyboard::getIns()->getPressingCount(KEY_INPUT_9) == 1) { _temp = 9; }
192
193 if (_temp >= 0)
194 {
195 if (m_input_number < 0)
196 {
197 m_input_number = _temp;
198 }
199 else
200 {
201 m_input_number *= 10;
202 m_input_number += _temp;
203 }
204 }
205}

208void GraphViewerGUIController::changeDisplayNodeIndex()
209{
210 if (mp_graph->size() == 0) return;
211
212 if (Keyboard::getIns()->getPressingCount(KEY_INPUT_DOWN) == 1)
213 {
214 (*mp_display_node_index)--;
215 }
216 else if (Keyboard::getIns()->getPressingCount(KEY_INPUT_UP) == 1)
217 {
218 (*mp_display_node_index)++;
219 }
220 else if (Keyboard::getIns()->getPressingCount(KEY_INPUT_RETURN) == 1)
221 {
222 (*mp_display_node_index) = m_input_number;
223 m_input_number = -1;
224 }
225 else if (Keyboard::getIns()->getPressingCount(KEY_INPUT_P) == 1)
226 {
227 if (mp_graph->size() > *mp_display_node_index)
228 {
229 (*mp_display_node_index) = mp_graph->at(*mp_display_node_index).parent_num;
230 }
231 }
232 else if (Keyboard::getIns()->getPressingCount(KEY_INPUT_LEFT) == 1 && m_childen_list.second.empty() == false)
233 {
234 m_display_children_list_index--;
235
236 if (m_display_children_list_index < 0) m_display_children_list_index = (int)m_childen_list.second.size() - 1;
237 m_display_children_list_index = m_display_children_list_index < 0 ? 0 : m_display_children_list_index;
238
239 (*mp_display_node_index) = m_childen_list.second.at(m_display_children_list_index);
240 }
241 else if (Keyboard::getIns()->getPressingCount(KEY_INPUT_RIGHT) == 1 && m_childen_list.second.empty() == false)
242 {
243 m_display_children_list_index++;
244
245 if (m_display_children_list_index >= m_childen_list.second.size()) m_display_children_list_index = 0;
246 (*mp_display_node_index) = m_childen_list.second.at(m_display_children_list_index);
247 }
248
249
250 if (*mp_display_node_index < 0)*mp_display_node_index = mp_graph->size() - 1;
251 else if (*mp_display_node_index >= mp_graph->size()) *mp_display_node_index = 0;
252}
