29void LegUpDownNodeCreator::create(const SNode& current_node, const int current_num, std::vector<SNode>* output_graph)
30{
31 //脚の遊脚・接地によって生じるとりうる重心をcomtypeとして仕分けている．(詳しくはComtype.hを参照)．まずは全てtrueにしておく．
32 bool is_able_type[ComType::COM_TYPE_NUM];
33
34 for (int i = 0; i < ComType::COM_TYPE_NUM; i++)
35 {
36 is_able_type[i] = true;
37 }
38
39 //重心が現在どこにあるか(前よりか真ん中か...)などのパラメータはこのcom patternで仕分けている．(詳しくはComtype.hを参照)．これを取得する．
40 int com_pattern = dl_leg::getComPatternState(current_node.leg_state);
41
42 //com patternよりとることができないcom typeを全てfalseにする．
43 ComType::checkAbleComTypeFromComPattern(com_pattern, is_able_type);
44
45
46
47 //次に脚が地面に接地可能か調べる．
48
49 bool is_groundable_leg[HexapodConst::LEG_NUM]; //脚が設置可能ならばtrueになる．既に接地しているならばtrueになる．
50 my_vec::SVector ground_pos[HexapodConst::LEG_NUM]; //脚が接地する座標．
51
52 for (int i = 0; i < HexapodConst::LEG_NUM; i++) { ground_pos[i] = current_node.leg_pos[i]; }
53
54 for (int i = 0; i < HexapodConst::LEG_NUM; i++)
55 {
56 if (dl_leg::isGrounded(current_node.leg_state, i))
57 {
58 //すでに接地している脚は接地可能に決まっているのでtrueにする．
59 is_groundable_leg[i] = true;
60 ground_pos[i] = current_node.leg_pos[i];
61 }
62 else
63 {
64 //現在遊脚中の脚は自身の脚状態で接地できるか検討する．
65 my_vec::SVector res_ground_pos;
66
67 if (isGroundableLeg(i, current_node, &res_ground_pos))
68 {
69 is_groundable_leg[i] = true; //接地可能にする．
70 ground_pos[i] = res_ground_pos;
71 }
72 else
73 {
74 is_groundable_leg[i] = false; //接地不可能にする．
75 ComType::checkAbleComTypeFromNotGroundableLeg(i, is_able_type); //接地不可能な脚によって，とれないcom typeを全てけす．
76 }
77 }
78 }
79
80
81 //子ノードを生成する．
82 for (int i = 0; i < ComType::COM_TYPE_NUM; i++)
83 {
84 //その重心タイプが可能であれば，
85 if (is_able_type[i])
86 {
87 SNode res_node = current_node;
88 res_node.changeNextNode(current_num, m_next_move);
89
90 //遊脚・接地を書き換える．
91 bool new_is_ground[HexapodConst::LEG_NUM] = {};
92 ComType::getGroundLegFromComType(i, new_is_ground);
93
94 for (int j = 0; j < HexapodConst::LEG_NUM; j++)
95 {
96 dl_leg::changeGround(res_node.leg_state, j, new_is_ground[j]);
97
98 if (new_is_ground[j])
99 {
100 res_node.leg_pos[j] = ground_pos[j];
101 res_node.leg_base_pos[j] = ground_pos[j];
102 }
103 else
104 {
105 res_node.leg_pos[j].x = 160 * HexapodConst::DEFAULT_LEG_ANGLE_COS[j];
106 res_node.leg_pos[j].y = 160 * HexapodConst::DEFAULT_LEG_ANGLE_SIN[j];
107 res_node.leg_pos[j].z = -10;
108 res_node.leg_base_pos[j].x = res_node.leg_pos[j].x;
109 res_node.leg_base_pos[j].y = res_node.leg_pos[j].y;
110 }
111 }
112
113 if (m_calclator.isAblePause(res_node))
114 {
115 //脚が地面についているならば追加せずに続行．
116 (*output_graph).push_back(res_node);
117 }
118
119 }
120
121 }

126bool LegUpDownNodeCreator::isGroundableLeg(const int now_leg_num, const SNode& current_node, my_vec::SVector* output_ground_pos)
127{
128 //for文の中のcontinueについては http://www9.plala.or.jp/sgwr-t/c/sec06-7.html を参照．ちなみに読みづらくなるので本当は使わないほうがいい．
129
130 if (mp_map == nullptr) { return false; } //マップがないときはfalseを返す．
131
132 //脚座標がdevide mapでどこに当たるか調べて，そのマスの2つ上と2つ下の範囲内を全て探索する．
133 const my_vec::SVector kGlobalLegbasePos = m_calclator.getGlobalLegBasePos(current_node, now_leg_num, false);
134
135 int max_x_dev = mp_map->getDevideMapNumX(kGlobalLegbasePos.x) + 2;
136 int min_x_dev = mp_map->getDevideMapNumX(kGlobalLegbasePos.x) - 2;
137 int max_y_dev = mp_map->getDevideMapNumY(kGlobalLegbasePos.y) + 2;
138 int min_y_dev = mp_map->getDevideMapNumY(kGlobalLegbasePos.y) - 2;
139
141 max_x_dev = (max_x_dev >= MapConst::LP_DIVIDE_NUM) ? MapConst::LP_DIVIDE_NUM - 1 : max_x_dev;
142 min_x_dev = (min_x_dev < 0) ? 0 : min_x_dev;
143 max_y_dev = (max_y_dev >= MapConst::LP_DIVIDE_NUM) ? MapConst::LP_DIVIDE_NUM - 1 : max_y_dev;
144 min_y_dev = (min_y_dev < 0) ? 0 : min_y_dev;
145
146
147 //devide map内を全探索して，現在の脚位置(離散化した物)に適した脚設置可能点が存在するか調べる．
148
149 my_vec::SVector candidate_pos; //現在の脚位置に合致する候補座標群．
150 bool is_candidate_pos = false; //候補座標が存在するかどうか．
151
152 //範囲内の点を全て調べる．
153 for (int x = min_x_dev; x < max_x_dev; x++)
154 {
155 for (int y = min_y_dev; y < max_y_dev; y++)
156 {
157 const int kPosNum = mp_map->getPointNumFromDevideMap(x, y);
158
159 for (int n = 0; n < kPosNum; n++)
160 {
161 my_vec::SVector map_point_pos = mp_map->getPosFromDevideMap(x, y, n); //脚設置可能点の座標を取り出す．
162 map_point_pos = m_calclator.convertLocalLegPos(current_node, map_point_pos, now_leg_num, false);
163
164 //脚位置を更新したノードを作成する．
165 SNode new_node = current_node;
166
167 new_node.leg_pos[now_leg_num] = map_point_pos;
168
169
170 //前の候補地点と比較して，より良い候補地点の時のみ実行すする
171 if (is_candidate_pos)
172 {
173 //反対方向をむいている場合は候補地点として採用しない．
174 if (new_node.leg_base_pos[now_leg_num].projectedXY().cross(candidate_pos.projectedXY()) * new_node.leg_base_pos[now_leg_num].projectedXY().cross(map_point_pos.projectedXY()) < 0)
175 {
176 continue;
177 }
178
179 //現在の脚位置と候補地点の間に障害物がある場合は候補地点として採用しない．
180 if (map_point_pos.projectedXY().cross(candidate_pos.projectedXY()) * map_point_pos.projectedXY().cross(new_node.leg_base_pos[now_leg_num].projectedXY()) < 0)
181 {
182 continue;
183 }
184 }
185
186 dl_leg::changeGround(new_node.leg_state, now_leg_num, true);
187
188 if (!m_calclator.isLegInRange(new_node, now_leg_num)) { continue; } //脚が範囲外ならば追加せずに続行．
189
190 //if (m_calclator.isLegInterfering(new_node)) { continue; } //脚が干渉しているならば追加せずに続行．
191
192 if (!isAbleLegPos(new_node, now_leg_num)) { continue; } //候補座標として，適していないならば追加せずに続行．
193
194 is_candidate_pos = true;
195 candidate_pos = map_point_pos;
196 }
197 }
198 }
199
200
201 //候補点を全列挙したのち，候補点が一つもなければfalse
202 if (!is_candidate_pos) { return false; }
203
204 //存在するなら，その中で最も適したものを結果として返し，true
205 (*output_ground_pos) = candidate_pos;
206
207 return true;
208}

211bool LegUpDownNodeCreator::isAbleLegPos(const SNode& _node, const int _leg_num)
212{
213 const int _leg_state = dl_leg::getLegState(_node.leg_state, _leg_num); //脚位置を取得(1〜7)
214
215 //まず最初に脚位置4のところにないか確かめる．
216 if ((_node.leg_base_pos[_leg_num] - _node.leg_pos[_leg_num]).lengthSquare() < my_math::squared(LEG_MARGIN))
217 {
218 if (_leg_state == 4) { return true; }
219 else { return false; }
220 }
221 else
222 {
223 if (_leg_state == 4) { return false; }
224 }
225
226 //脚位置4と比較して前か後ろか
227 if (_node.leg_base_pos[_leg_num].projectedXY().cross(_node.leg_pos[_leg_num].projectedXY()) * _node.leg_pos[_leg_num].projectedXY().cross({ 1,0 }) > 0)
228 {
229 //前
230
231 if (_leg_state == 1 || _leg_state == 2 || _leg_state == 3)
232 {
233 return false;
234 }
235 }
236 else
237 {
238 //後ろ
239
240 if (_leg_state == 7 || _leg_state == 6 || _leg_state == 5)
241 {
242 return false;
243 }
244 }
245
246
247 //脚位置4と比較して上か下か
248 if (_leg_state == 1 || _leg_state == 5)
249 {
250 //脚位置4と比較して下
251 if (_node.leg_base_pos[_leg_num].z - HIGH_MARGIN >= _node.leg_pos[_leg_num].z)
252 {
253 return true;
254 }
255 }
256 else if (_leg_state == 3 || _leg_state == 7)
257 {
258 //脚位置4と比較して上
259 if (_node.leg_base_pos[_leg_num].z + HIGH_MARGIN <= _node.leg_pos[_leg_num].z)
260 {
261 return true;
262 }
263 }
264 else
265 {
266 //脚位置4と同じくらい
267 if (std::abs(_node.leg_base_pos[_leg_num].z - _node.leg_pos[_leg_num].z) <= HIGH_MARGIN)
268 {
269 return true;
270 }
271 }
272
273 return false;
274}
