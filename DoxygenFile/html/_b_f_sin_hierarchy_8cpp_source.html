76void bfs_in_hierarchy(char start, char* prev, char* cost, bool* visited)
77{
78 bool graph_mask[36] = {0};
79 bool updating_graph_mask[36] = {0};
80
81 for (int i = 0; i < 36; i++)
82 { //初期化
83 graph_mask[i] = false;
84 updating_graph_mask[i] = false;
85 }
86
87 graph_mask[start] = true; //スタート位置は発見済みに(親ノードは発見済み)
88 visited[start] = true;//親ノードは訪問済み
89
90 for (int i = 0; i < 36; ++i)
91 {
92 prev[i] = -1; //初期化
93 cost[i] = -1;
94 }
95
96 prev[start] = -1;
97 cost[start] = 0;
98
99 //踏みかえ動作探索（踏みかえは2回でできる）
100 for (int i = 0; i < 2; ++i)
101 {
102 for (int j = 0; j < 36; ++j)
103 {
104 //現在いるノード（親ノードの支持脚番号なったら、）
105 if (graph_mask[j])
106 {
107 graph_mask[j] = false;
108
109 for (int k = edge_index[j]; k < edge_index[j+1]; ++k)
110 {
111 char id = edge_list[k];
112
113 //ノードが発見済みでなければ
114 if (!visited[id])
115 {
116 cost[id] = cost[j] + 1; //コスト＝親の深さ＋1
117 prev[id] = j; //親ノード
118 graph_mask[id] = true; //発見したノード
119 visited[id] = true; //探索済み

121 }

123 }
