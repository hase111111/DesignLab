180 inline SVector rotVector(const SVector& _v, const SRotator& _rot)
181 {
182 float _x = cos(_rot.yaw) * cos(_rot.pitch) * _v.x
183 + (cos(_rot.yaw) * sin(_rot.pitch) * sin(_rot.roll) - sin(_rot.yaw) * cos(_rot.roll)) * _v.y
184 + (cos(_rot.yaw) * sin(_rot.pitch) * cos(_rot.roll) + sin(_rot.yaw) * sin(_rot.roll)) * _v.z;
185
186 float _y = sin(_rot.yaw) * cos(_rot.pitch) * _v.x
187 + (sin(_rot.yaw) * sin(_rot.pitch) * sin(_rot.roll) + cos(_rot.yaw) * cos(_rot.roll)) * _v.y
188 + (sin(_rot.yaw) * sin(_rot.pitch) * cos(_rot.roll) - cos(_rot.yaw) * sin(_rot.roll)) * _v.z;
189
190 float _z = -sin(_rot.pitch) * _v.x + cos(_rot.pitch) * sin(_rot.roll) * _v.y + cos(_rot.pitch) * cos(_rot.roll) * _v.z;
191
192 return SVector(_x, _y, _z);

206 //Y(z軸右回り)→P(y軸右回り)→R(x軸右回り)
207 //VRot_R(my_vec::VECTOR In, my_vec::VECTOR center, float thP, float thR, float thY)
208 //R(y軸左回り)→P(x軸左回り)→Y(z軸左回り)
210 //ベクトルの3Dローテーション オイラー角（テイト・ブライアン角）Y-X-Z
211 __inline SVector VRot(const SVector In, const SVector center, const float thP, const float thR, const float thY)
212 {
213 SVector ans, buf;
214 buf = In - center;
215 ans.x = (cos(thR) * cos(thY) + sin(thR) * sin(thP) * sin(thY)) * buf.x +
216 (cos(thY) * sin(thR) * sin(thP) - cos(thR) * sin(thY)) * buf.y +
217 (cos(thP) * sin(thR)) * buf.z;
218
219 ans.y = cos(thP) * sin(thY) * buf.x +
220 cos(thP) * cos(thY) * buf.y +
221 (-sin(thP)) * buf.z;
222
223 ans.z = (cos(thR) * sin(thP) * sin(thY) - cos(thY) * sin(thR)) * buf.x +
224 (cos(thR) * cos(thY) * sin(thP) + sin(thR) * sin(thY)) * buf.y +
225 (cos(thR) * cos(thP)) * buf.z;
226
227 return ans + center;
228 }
