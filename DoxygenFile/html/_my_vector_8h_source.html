168 inline SVector rotVector(const SVector& _v, const SRotator& _rot)
169 {
170 float _x = cos(_rot.yaw) * cos(_rot.pitch) * _v.x
171 + (cos(_rot.yaw) * sin(_rot.pitch) * sin(_rot.roll) - sin(_rot.yaw) * cos(_rot.roll)) * _v.y
172 + (cos(_rot.yaw) * sin(_rot.pitch) * cos(_rot.roll) + sin(_rot.yaw) * sin(_rot.roll)) * _v.z;
173
174 float _y = sin(_rot.yaw) * cos(_rot.pitch) * _v.x
175 + (sin(_rot.yaw) * sin(_rot.pitch) * sin(_rot.roll) + cos(_rot.yaw) * cos(_rot.roll)) * _v.y
176 + (sin(_rot.yaw) * sin(_rot.pitch) * cos(_rot.roll) - cos(_rot.yaw) * sin(_rot.roll)) * _v.z;
177
178 float _z = -sin(_rot.pitch) * _v.x + cos(_rot.pitch) * sin(_rot.roll) * _v.y + cos(_rot.pitch) * cos(_rot.roll) * _v.z;
179
180 return SVector(_x, _y, _z);
