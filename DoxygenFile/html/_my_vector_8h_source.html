140 inline SVector rotVector(const SVector& _v, const SRotator& _rot)
141 {
142 float _x = cos(_rot.yaw) * cos(_rot.pitch) * _v.x
143 + (cos(_rot.yaw) * sin(_rot.pitch) * sin(_rot.roll) - sin(_rot.yaw) * cos(_rot.roll)) * _v.y
144 + (cos(_rot.yaw) * sin(_rot.pitch) * cos(_rot.roll) + sin(_rot.yaw) * sin(_rot.roll)) * _v.z;
145
146 float _y = sin(_rot.yaw) * cos(_rot.pitch) * _v.x
147 + (sin(_rot.yaw) * sin(_rot.pitch) * sin(_rot.roll) + cos(_rot.yaw) * cos(_rot.roll)) * _v.y
148 + (sin(_rot.yaw) * sin(_rot.pitch) * cos(_rot.roll) - cos(_rot.yaw) * sin(_rot.roll)) * _v.z;
149
150 float _z = -sin(_rot.pitch) * _v.x + cos(_rot.pitch) * sin(_rot.roll) * _v.y + cos(_rot.pitch) * cos(_rot.roll) * _v.z;
151
152 return SVector(_x, _y, _z);

167 //VRot_R(my_vec::VECTOR In, my_vec::VECTOR center, float thP, float thR, float thY)
168 //R(y軸左回り)→P(x軸左回り)→Y(z軸左回り)
170 //ベクトルの3Dローテーション オイラー角（テイト・ブライアン角）Y-X-Z
171 __inline SVector VRot(const SVector In, const SVector center, const float thP, const float thR, const float thY)
172 {
173 SVector ans, buf;
174 buf = In - center;
175 ans.x = (cos(thR) * cos(thY) + sin(thR) * sin(thP) * sin(thY)) * buf.x +
176 (cos(thY) * sin(thR) * sin(thP) - cos(thR) * sin(thY)) * buf.y +
177 (cos(thP) * sin(thR)) * buf.z;
178
179 ans.y = cos(thP) * sin(thY) * buf.x +
180 cos(thP) * cos(thY) * buf.y +
181 (-sin(thP)) * buf.z;
182
183 ans.z = (cos(thR) * sin(thP) * sin(thY) - cos(thY) * sin(thR)) * buf.x +
184 (cos(thR) * cos(thY) * sin(thP) + sin(thR) * sin(thY)) * buf.y +
185 (cos(thR) * cos(thP)) * buf.z;
186
187 return ans + center;
188 }
