284 void flatten_and_destroy() noexcept
285 {
286 while (!elements_.empty())
287 {
288 value_type current = std::move(elements_.back());
289 elements_.pop_back();
290 switch (current.storage_kind())
291 {
292 case json_storage_kind::array_value:
293 {
294 for (auto&& item : current.array_range())
295 {
296 if (item.size() > 0) // non-empty object or array
297 {
298 elements_.push_back(std::move(item));
299 }
300 }
301 current.clear();
302 break;
303 }
304 case json_storage_kind::object_value:
305 {
306 for (auto&& kv : current.object_range())
307 {
308 if (kv.value().size() > 0) // non-empty object or array
309 {
310 elements_.push_back(std::move(kv.value()));
311 }
312 }
313 current.clear();
314 break;
315 }
316 default:
317 break;
318 }
319 }
320 }
