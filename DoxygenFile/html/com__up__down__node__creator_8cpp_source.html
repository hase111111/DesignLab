31void ComUpDownNodeCreator::create(const SNode& current_node, const int current_num, std::vector<SNode>* output_graph)
32{
33 //重心を最も高くあげることのできる位置と，最も低く下げることのできる位置を求める．グローバル座標で Zの位置．
34 //マップを確認して地面の最高点を求め，そこからMAX_RANGE，MIN_RANGEの分だけ離す．
35
36
37 //マップの最大z座標を求める．
38 const int kMapX = mp_map->getDevideMapNumX(current_node.global_center_of_mass.x);
39 const int kMapY = mp_map->getDevideMapNumY(current_node.global_center_of_mass.y);
40 const float kMapHighestZ = mp_map->getTopZFromDevideMap(kMapX, kMapY);
41
42 //ロボットの重心の最も低く下げることのできるz座標と，高くあげることができるz座標を求める．どちらもグローバル座標．
43 float highest_body_zpos = kMapHighestZ + HexapodConst::VERTICAL_MAX_RANGE;
44 float lowest_body_zpos = kMapHighestZ + HexapodConst::VERTICAL_MIN_RANGE;
45
46
47 // 最も高い地点を修正する．
48
49 for (int i = 0; i < HexapodConst::LEG_NUM; i++)
50 {
51 //接地している脚についてのみ考える．
52 if (dl_leg::isGrounded(current_node.leg_state, i))
53 {
54 //三平方の定理を使って，脚接地地点から重心位置をどれだけ上げられるか考える．
55 const float edge_c = HexapodConst::FEMUR_LENGTH + HexapodConst::TIBIA_LENGTH - MARGIN;
56 const float edge_b = current_node.leg_pos[i].projectedXY().length() - HexapodConst::COXA_LENGTH;
57
58 const float edge_a = sqrt(my_math::squared(edge_c) - my_math::squared(edge_b));
59
60 //接地脚の最大重心高さの中から一番小さいものを全体の最大重心位置として記録する．_aは脚の接地点からどれだけ上げられるかを表しているので，グローバル座標に変更する．
61 highest_body_zpos = (std::min)(edge_a + current_node.global_center_of_mass.z + current_node.leg_pos[i].z, highest_body_zpos);
62 }
63 }
64
65
66 //ノードを追加する．
67 pushNodeByMaxAndMinPosZ(current_node, current_num, highest_body_zpos, lowest_body_zpos, output_graph);
68}

71void ComUpDownNodeCreator::pushNodeByMaxAndMinPosZ(const SNode& current_node, const int current_num, const float high, const float low, std::vector<SNode>* output_graph)
72{
73 //重心を変化させたものを追加する．変化量が一番少ないノードは削除する．
74 {
75 //最大と最小の間を分割する．
76 const float kDivZ = (high - low) / (float)DISCRETIZATION;
77
78 //現在の重心との差分が一番小さいものを探す．
79 float dif_min = 100000.0f;
80 int dif_min_index = -1;
81
82 //分割した分新しいノードを追加する．
83 for (int i = 0; i < DISCRETIZATION + 1; i++)
84 {
85 SNode new_node = current_node;
86
87 //重心の位置を変更する．
88 my_vec::SVector new_com = current_node.global_center_of_mass;
89 new_com.z = low + kDivZ * i;
90
91 new_node.changeGlobalCenterOfMass(new_com, true);
92
93 if (dif_min > abs(current_node.global_center_of_mass.z - new_node.global_center_of_mass.z))
94 {
95 dif_min = abs(current_node.global_center_of_mass.z - new_node.global_center_of_mass.z);
96 dif_min_index = i;
97 }
98
99 //current_numを親とする，新しいノードに変更する
100 new_node.changeNextNode(current_num, m_next_move);
101
102 //ノードを追加する．
103 (*output_graph).emplace_back(new_node);
104 }
105
106 //一番差分が小さくものを消す
107 if (dif_min_index >= 0) { (*output_graph).erase((*output_graph).begin() + dif_min_index); }
108 }
109
110
111 //重心の変化が一切ないものを追加する．
112 {
113 SNode same_node = current_node;
114
115 same_node.changeNextNode(current_num, m_next_move);
116
117 (*output_graph).emplace_back(same_node);
118 }
119}
