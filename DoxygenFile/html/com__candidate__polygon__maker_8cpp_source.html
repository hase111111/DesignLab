7void ComCandidatePolygonMaker::makeCandidatePolygon(const SNode& node, std::pair<my_vec::SPolygon2, ComType::EComPattern> output_poly[MAKE_POLYGON_NUM]) const
8{
9 my_vec::SVector2 leg_pos_xy[HexapodConst::LEG_NUM]; //XY平面に射影した脚位置を算出する
10
11 for (int i = 0; i < HexapodConst::LEG_NUM; i++)
12 {
13 leg_pos_xy[i] = m_calclator.getGlobalLegPos(node, i, false).projectedXY(); //脚位置(グローバル座標)をXY平面に射影する
14 }
15
16 //中心を囲むように4角形を作成する
17 makeCandidateBox(leg_pos_xy, 0, &output_poly[0].first); output_poly[0].second = ComType::EComPattern::FRONT_LEFT;
18 makeCandidateBox(leg_pos_xy, 1, &output_poly[1].first); output_poly[1].second = ComType::EComPattern::BACK_LEFT;
19 makeCandidateBox(leg_pos_xy, 2, &output_poly[2].first); output_poly[2].second = ComType::EComPattern::BACK;
20 makeCandidateBox(leg_pos_xy, 3, &output_poly[3].first); output_poly[3].second = ComType::EComPattern::BACK_RIGHT;
21 makeCandidateBox(leg_pos_xy, 4, &output_poly[4].first); output_poly[4].second = ComType::EComPattern::FRONT_RIGHT;
22 makeCandidateBox(leg_pos_xy, 5, &output_poly[5].first); output_poly[5].second = ComType::EComPattern::FRONT;
23
24 //中心に3角形を作成する
25 makeCandidateTriangle(leg_pos_xy, &output_poly[6].first, &output_poly[6].second);
26

39 //全ての多角形をcoutで出力する
40 if (DO_DEBUG_PRINT)

49void ComCandidatePolygonMaker::makeCandidateBox(const my_vec::SVector2 leg_pos[HexapodConst::LEG_NUM], const int start_leg_num, my_vec::SPolygon2* output_poly) const
50{
51 //脚位置を線で結ぶ．この交点から重心候補地点が存在する多角形を求める
52 my_vec::SLine2 leg_line_02(leg_pos[(start_leg_num + 0) % HexapodConst::LEG_NUM], leg_pos[(start_leg_num + 2) % HexapodConst::LEG_NUM]);
53 my_vec::SLine2 leg_line_03(leg_pos[(start_leg_num + 0) % HexapodConst::LEG_NUM], leg_pos[(start_leg_num + 3) % HexapodConst::LEG_NUM]);
54 my_vec::SLine2 leg_line_14(leg_pos[(start_leg_num + 1) % HexapodConst::LEG_NUM], leg_pos[(start_leg_num + 4) % HexapodConst::LEG_NUM]);
55 my_vec::SLine2 leg_line_15(leg_pos[(start_leg_num + 1) % HexapodConst::LEG_NUM], leg_pos[(start_leg_num + 5) % HexapodConst::LEG_NUM]);
56 my_vec::SLine2 leg_line_25(leg_pos[(start_leg_num + 2) % HexapodConst::LEG_NUM], leg_pos[(start_leg_num + 5) % HexapodConst::LEG_NUM]);
57
58 //交点(intersection)を求める
59 my_vec::SVector2 intersection_02_14 = leg_line_02.getIntersection(leg_line_14);
60 my_vec::SVector2 intersection_02_15 = leg_line_02.getIntersection(leg_line_15);
61 my_vec::SVector2 intersection_03_14 = leg_line_03.getIntersection(leg_line_14);
62 my_vec::SVector2 intersection_03_15 = leg_line_03.getIntersection(leg_line_15);
63
64 //中心と0番の脚位置を結んだ線分を求める
65 my_vec::SLine2 leg_line_0_center(leg_pos[(start_leg_num + 0) % HexapodConst::LEG_NUM], intersection_03_14);
66
67 //多角形生成
68
69 (*output_poly).reset();
70
71 if (leg_line_0_center.hasIntersection(leg_line_25))
72 {
73 //交点がある場合，5角形の多角形を作成する
74 my_vec::SVector2 intersection_03_25 = leg_line_03.getIntersection(leg_line_25);
75 my_vec::SVector2 intersection_14_25 = leg_line_14.getIntersection(leg_line_25);
76
77 (*output_poly).addVertexCheckForDuplicates(intersection_03_15);
78 (*output_poly).addVertexCheckForDuplicates(intersection_02_15);
79 (*output_poly).addVertexCheckForDuplicates(intersection_02_14);
80 (*output_poly).addVertexCheckForDuplicates(intersection_14_25);
81 (*output_poly).addVertexCheckForDuplicates(intersection_03_25);
82 }
83 else
84 {
85 //交点がない場合，既に求めた4点で，4角形の多角形を作成する
86 (*output_poly).addVertex(intersection_03_15);
87 (*output_poly).addVertex(intersection_02_15);
88 (*output_poly).addVertex(intersection_02_14);
89 (*output_poly).addVertex(intersection_03_14);
90 }
91}

103
104 //三角形を作成する．

113 }

116 (*out_com_pattern) = ComType::EComPattern::CENTER_FRONT;
