113void phantomxCommander::sendEndPoints(int movingTime)
114{
115 int command_length; //PCからPhantomXに送るデータの長さ[Byte]
116 int writeErrorFlag;
117 DWORD errorCode;
118 command[0]=0xfe; //データの始まり
119 command[1]=0xff; //
120 command[2]=write_serial; //ASCIIの'w'　書き込みの合図
121 command[3]=movingTime & 0xff; //脚先移動時間の下位8bit
122 command[4]=movingTime >> 8 & 0xff; //脚先移動時間の上位8bit
123 command[5]=LEG_COUNT * 7 + 1; //書き込むコマンドの長さ。（脚の数*（ID(1Byte)+x座標(2Byte)+y座標(2Byte)+z座標(2Byte)) + CheckSum2(1Byte))
124 command[6]=((command[2]+command[3]+command[4]+command[5]) ^ 0xff) & 0xff; //CheckSum1
125 for(int i = 0; i< LEG_COUNT; i++)
126 {
127 command[i*7+7] = i; //脚のID
128 command[i*7+8] = (int)myRound(endpoints[i].x) & 0xff; //脚先位置のx座標の下位8bit
129 command[i*7+9] = (int)myRound(endpoints[i].x) >> 8 & 0xff; //脚先位置のx座標の上位8bit
130 command[i*7+10] = (int)myRound(endpoints[i].y) & 0xff; //脚先位置のy座標の下位8bit
131 command[i*7+11] = (int)myRound(endpoints[i].y) >> 8 & 0xff; //脚先位置のy座標の上位8bit
132 command[i*7+12] = (int)myRound(endpoints[i].z) & 0xff; //脚先位置のz座標の下位8bit
133 command[i*7+13] = (int)myRound(endpoints[i].z) >> 8 & 0xff; //脚先位置のz座標の上位8bit
134 }
135 command_length = command[5] + 7;
136
137 int commandsum = 0;
138
139 for(int k=2;k<command_length-1;k++)commandsum+=command[k];
140 command[command_length-1]=(255 - (commandsum)%256);
141
142 for(int i=0;i<command_length;i++)
143 {
144 writeErrorFlag = WriteFile( h, &command[i], 1, &nn, 0 ); // シリアルポートに対する書き込み
145 //printf("%d\t%x\n",i,command[i]);
146 if(!writeErrorFlag)
147 {
148 errorCode = GetLastError();
149 std::cout << "Write file error! : " << errorCode << "\n";
150 }
151 }
152 std::cout << "I had sent endpoints.\n";
153 presentTime = movingTime;
154}
