138 //回転させたベクトルを返す．
139 inline SVector rotVector(const SVector& _v, const SRotator& _rot)
140 {
141 float _x = cos(_rot.yaw) * cos(_rot.pitch) * _v.x
142 + (cos(_rot.yaw) * sin(_rot.pitch) * sin(_rot.roll) - sin(_rot.yaw) * cos(_rot.roll)) * _v.y
143 + (cos(_rot.yaw) * sin(_rot.pitch) * cos(_rot.roll) + sin(_rot.yaw) * sin(_rot.roll)) * _v.z;
144
145 float _y = sin(_rot.yaw) * cos(_rot.pitch) * _v.x
146 + (sin(_rot.yaw) * sin(_rot.pitch) * sin(_rot.roll) + cos(_rot.yaw) * cos(_rot.roll)) * _v.y
147 + (sin(_rot.yaw) * sin(_rot.pitch) * cos(_rot.roll) - cos(_rot.yaw) * sin(_rot.roll)) * _v.z;
148
149 float _z = -sin(_rot.pitch) * _v.x + cos(_rot.pitch) * sin(_rot.roll) * _v.y + cos(_rot.pitch) * cos(_rot.roll) * _v.z;
150
151 return SVector(_x, _y, _z);
152 }

210 //座標系はmainの座標系でPRYそれぞれに角度を入れたときの回転は
211 //VRot(myvector::VECTOR In, myvector::VECTOR center, float thP, float thR, float thY)
212 //Y(z軸左回り)→P(x軸左回り)→R(y軸左回り)
213 //VRot(const myvector::VECTOR &In, float thP, float thR,float thY )
214 //Y(z軸右回り)→P(y軸右回り)→R(x軸右回り)
215 //VRot_R(myvector::VECTOR In, myvector::VECTOR center, float thP, float thR, float thY)
216 //R(y軸左回り)→P(x軸左回り)→Y(z軸左回り)
218 //ベクトルの3Dローテーション オイラー角（テイト・ブライアン角）Y-X-Z
219 __inline SVector VRot(const SVector In, const SVector center, const float thP, const float thR, const float thY)
220 {
221 SVector ans, buf;
222 buf = In - center;
223 ans.x = (cos(thR) * cos(thY) + sin(thR) * sin(thP) * sin(thY)) * buf.x +
224 (cos(thY) * sin(thR) * sin(thP) - cos(thR) * sin(thY)) * buf.y +
225 (cos(thP) * sin(thR)) * buf.z;
226
227 ans.y = cos(thP) * sin(thY) * buf.x +
228 cos(thP) * cos(thY) * buf.y +
229 (-sin(thP)) * buf.z;
230
231 ans.z = (cos(thR) * sin(thP) * sin(thY) - cos(thY) * sin(thR)) * buf.x +
232 (cos(thR) * cos(thY) * sin(thP) + sin(thR) * sin(thY)) * buf.y +
233 (cos(thR) * cos(thP)) * buf.z;
234
235 return ans + center;
236 }

257 __inline SVector VRot_R(SVector In, SVector center, const float thP, const float thR, const float thY)
258 {
259
260 SVector ans, buf;
261 buf = In - center;
262 ans.x = (cos(thY) * cos(thR) - sin(thY) * sin(thP) * sin(thR)) * buf.x +
263 (-cos(thP) * sin(thY)) * buf.y +
264 (cos(thR) * sin(thY) * sin(thP) + cos(thY) * sin(thR)) * buf.z;

267 (cos(thY) * cos(thP)) * buf.y +
268 (sin(thY) * sin(thR) - cos(thY) * cos(thR) * sin(thP)) * buf.z;
269
270
271 ans.z = -cos(thP) * sin(thR) * buf.x +
272 sin(thP) * buf.y +
273 cos(thP) * cos(thR) * buf.z;
