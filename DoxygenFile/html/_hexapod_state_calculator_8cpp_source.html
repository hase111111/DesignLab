34void HexapodStateCalclator::calclateJointPos(const SNode& _node)
35{
36 // 逆運動学的にジョイントの場所を計算する．
37 //ノードの脚位置は正しい場所にあるという前提のもと計算するので，めちゃくちゃな値が代入されているとうまく動作しない．
38 //チェックする機能を付けると重くなるので，そもそもそんなノードを生成しないように注意する．
39
40 using namespace my_math;
41
42 for (int i = 0; i < HexapodConst::LEG_NUM; i++)
43 {
44 const float _coxa_joint_angle = atan2(_node.leg_pos[i].y, _node.leg_pos[i].x);
45
46 m_local_femurjoint_pos[i] = my_vec::SVector(HexapodConst::COXA_LENGTH * cos(_coxa_joint_angle), HexapodConst::COXA_LENGTH * sin(_coxa_joint_angle), 0);
47
48
49 const float _L = std::sqrt(squared(_node.leg_pos[i].x - m_local_femurjoint_pos[i].x) + squared(_node.leg_pos[i].y - m_local_femurjoint_pos[i].y)); //脚先から第一関節までの長さ．
50 const float _leg_to_fumur_len = std::sqrt(squared(_L) + squared(_node.leg_pos[i].z));
51
52 const float _s1 = squared(_leg_to_fumur_len) + squared(HexapodConst::TIBIA_LENGTH) - squared(HexapodConst::FEMUR_LENGTH);
53 const float _s2 = 2 * HexapodConst::TIBIA_LENGTH * _leg_to_fumur_len;
54
55 const float _fumur_joint_angle = -(std::acos(_s1 / _s2) + std::atan(-_node.leg_pos[i].z / _L));
56
57 m_local_tibiajoint_pos[i] = _node.leg_pos[i] -
58 my_vec::SVector(HexapodConst::TIBIA_LENGTH * cos(_coxa_joint_angle) * cos(_fumur_joint_angle),
59 HexapodConst::TIBIA_LENGTH * sin(_coxa_joint_angle) * cos(_fumur_joint_angle),
60 HexapodConst::TIBIA_LENGTH * sin(_fumur_joint_angle));
61
62 if (abs((m_local_femurjoint_pos[i] - m_local_tibiajoint_pos[i]).length() - HexapodConst::FEMUR_LENGTH) > my_math::ALLOWABLE_ERROR)
63 {
64 const float _fumur_joint_angle = -(-std::acos(_s1 / _s2) + std::atan(-_node.leg_pos[i].z / _L));
65
66 m_local_tibiajoint_pos[i] = _node.leg_pos[i] -
67 my_vec::SVector(HexapodConst::TIBIA_LENGTH * cos(_coxa_joint_angle) * cos(_fumur_joint_angle),
68 HexapodConst::TIBIA_LENGTH * sin(_coxa_joint_angle) * cos(_fumur_joint_angle),
69 HexapodConst::TIBIA_LENGTH * sin(_fumur_joint_angle));
70 }
71 }
72

86{

88
89 for (int _z = 0; _z < MAX_DIF_Z; _z++)
90 {
91 float _max_r = 0;
92 float _min_r = 0;
93
94 for (float _x = HexapodConst::COXA_LENGTH; _x < HexapodConst::COXA_LENGTH + HexapodConst::FEMUR_LENGTH + HexapodConst::TIBIA_LENGTH; _x++)
95 {
96 my_vec::SVector _tmp_leg((float)_x, 0, -(float)_z);
97
98 // 以下の三変数を辺とする三角形が成立するか調べる．
99 float _a = HexapodConst::TIBIA_LENGTH;
100 float _b = HexapodConst::FEMUR_LENGTH;
101 float _c = sqrt(squared(_tmp_leg.x - HexapodConst::COXA_LENGTH) + squared(_tmp_leg.z));
102
103 bool _is_vaild_triangle = true;
104 if (_a + _b < _c)_is_vaild_triangle = false;
105 if (_a + _c < _b)_is_vaild_triangle = false;
106 if (_c + _b < _a)_is_vaild_triangle = false;
107
108 //成立するのならば，そこに脚を伸ばすことができるはず．
109 if (_is_vaild_triangle)
110 {
111 if (_min_r == 0)_min_r = _x;
112 _max_r = _x;
113 }
114
115 }
116
117 m_leg_max_r[_z] = _max_r;
118 m_leg_min_r[_z] = _min_r;
119 }

145{
146 //重心を原点とした，座標系において，脚の干渉を調べる．
147
148 //脚の干渉を調べる．
149 my_vec::SVector2 _leg_pos[HexapodConst::LEG_NUM];
150 my_vec::SVector2 _joint_pos[HexapodConst::LEG_NUM];
151
152 for (int i = 0; i < HexapodConst::LEG_NUM; i++)

247float HexapodStateCalclator::calculateStaticMargin(const SNode& node) const
248{
249 //重心を原点とした座標系で，脚の位置を計算する．
250 // std::min をカッコで囲んでいるのは，マクロの min と被るため．(std::min) と書くと名前が衝突しない
251
252 std::vector<my_vec::SVector2> leg_pos;
253
254 //接地脚のみ追加する
255 for (int i = 0; i < HexapodConst::LEG_NUM; i++)
256 {
257 if (dl_leg::isGrounded(node.leg_state, i) == true)
258 {
259 leg_pos.push_back(node.leg_pos[i].projectedXY() + getLocalCoxaJointPos(i).projectedXY());
260 }
261 }
262
263 float min_margin = 1000000;
264
265 for (int i = 0; i < leg_pos.size(); i++)
266 {
267 my_vec::SVector2 i_to_i_plus_1 = leg_pos.at((i + 1) % leg_pos.size()) - leg_pos.at(i);
268 i_to_i_plus_1.normalized();
269
270 my_vec::SVector2 i_to_com = my_vec::SVector2{ 0,0 } - leg_pos.at(i);
271
272 min_margin = (std::min)(min_margin, i_to_com.cross(i_to_i_plus_1));
273 }
274
275 return min_margin;
