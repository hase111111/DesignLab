13void MapCreator::create(const EMapCreateMode mode, const int option, const bool do_output, std::vector<my_vec::SVector>* map_data)
14{
15 switch (mode)
16 {
17 case EMapCreateMode::FLAT:
18
19 createFlatMap(map_data);
20 break;
21
22 case EMapCreateMode::VERTICAL_STRIPE:
23
24 createVerticalStripeMap(map_data);
25 break;
26
27 case EMapCreateMode::HORIZONTAL_STRIPE:
28
29 createHorizontalStripeMap(map_data);
30 break;
31
32 case EMapCreateMode::DIAGONAL_STRIPE:
33
34 createDiagonalStripeMap(map_data);
35 break;
36
37 case EMapCreateMode::MESH:
38
39 createMeshMap(map_data);
40 break;
41
42 case EMapCreateMode::LATTICE_POINT:
43
44 createLatticePointMap(map_data);
45 break;
46
47 case EMapCreateMode::READ_FROM_FILE:
48
49 //ファイルからマップを生成する．
50 if (!createMapFromFile(map_data))
51 {
52 //読み込みに失敗したら，平面のマップを生成する．
53 createFlatMap(map_data);
54 }
55 break;
56
57 default:
58
59 //異常な値が入力されたら，平面のマップを生成する．
60 createFlatMap(map_data);
61 break;
62 }
63
64 //オプション指定に基づき，Z座標を変更する
65 if (option & OPTION_PERFORATED && mode != EMapCreateMode::READ_FROM_FILE)
66 {
67 //穴あき地形にする．
68 changeMapPerforated(map_data);
69 }
70
71 if (option & OPTION_STEP && mode != EMapCreateMode::READ_FROM_FILE)
72 {
73 //階段状にする．
74 changeMapStep(map_data);
75 }
76
77 if (option & OPTION_SLOPE && mode != EMapCreateMode::READ_FROM_FILE)
78 {
79 //坂道にする．
80 changeMapSlope(map_data);
81 }
82
83 if (option & OPTION_TILT && mode != EMapCreateMode::READ_FROM_FILE)
84 {
85 //坂道にする．
86 changeMapTilt(map_data);
87 }
88
89 if (option & OPTION_ROUGH && mode != EMapCreateMode::READ_FROM_FILE)
90 {
91 //デコボコにする．
92 changeMapRough(map_data);
93 }
94
95 //ファイルへ出力する
96 if (do_output)
97 {
98 writeMapToFile(*map_data);
99 }
100
101}

288bool MapCreator::createMapFromFile(std::vector<my_vec::SVector>* map)
289{
290 //マップデータを初期化する
291 (*map).clear();
292
293 std::ifstream ifs(MapConst::INPUT_FILE_NAME);//読み込むファイル
294 std::string line;//ファイルの行
295
296 //ファイルを開くことに失敗したら
297 if (!ifs.is_open())
298 {
299 std::cout << "マップデータファイルが開けません．" << std::endl;
300 return false;
301 }
302
303 // 1行目には数値がないため，読み込まない
304 bool is_first_line = true;
305
306 // 1行ごとに読み込む
307 while (std::getline(ifs, line))
308 {
309 std::istringstream stream(line);
310 std::string field;
311 std::vector<std::string> result;
312
313 while (std::getline(stream, field, ','))
314 {
315 result.push_back(field); //","ごとにresultにプッシュバック．この時点では文字として入力される
316 }
317
318 if (!is_first_line)
319 {
320 //stringをfloatに変換して代入
321 my_vec::SVector _pos(stof(result.at(1)), stof(result.at(2)), stof(result.at(3)));
322 (*map).push_back(_pos);
323 }
324 else { is_first_line = false; }
325
326 }
327
328 ifs.close();
329
330 return true;
331}

333void MapCreator::changeMapPerforated(std::vector<my_vec::SVector>* map)
334{
335 //厳密にホール率に合わせるために，まずはマップをSTRIPE_INTERVALにあわせて区切って，全部で何マスあるか調べる．
336 const int kCellNumX = (MapConst::MAP_MAX_FORWARD - MapConst::MAP_START_ROUGH) / MapConst::FOOT_HOLD_XY_DIST / MapConst::STRIPE_INTERVAL;
337 const int kCellNumY = (MapConst::MAP_MAX_HORIZONTAL - MapConst::MAP_MIN_HORIZONTAL) / MapConst::FOOT_HOLD_XY_DIST / MapConst::STRIPE_INTERVAL;
338 const int kCellNum = kCellNumX * kCellNumY;
339
340 //マスの数だけ要素を持つvectorを用意する．値は全てfalseで初期化する．
341 std::vector<bool> do_perforated(kCellNum, false);
342
343 //ホール率に合わせて，値をtrueに変更する
344 const int kHoleNum = kCellNum * MapConst::HOLE_RATE / 100;
345
346 for (int i = 0; i < kHoleNum; i++)
347 {
348 do_perforated.at(i) = true;
349 }
350
351 //ランダムなホールにするために要素の順番をシャッフルする．
352 std::shuffle(std::begin(do_perforated), std::end(do_perforated), std::default_random_engine());
353
354 //マップに穴をあける
355 for (auto itr = (*map).begin(); itr != (*map).end();)
356 {
357 //待機場所の外に対してのみ作業をする
358 if ((*itr).x < MapConst::MAP_START_ROUGH)
359 {
360 itr++;
361 continue;
362 }
363
364 //マスで区切るとどこに位置するかを調べる．
365 const int kCellPosX = (int)((*itr).x - MapConst::MAP_START_ROUGH) / MapConst::FOOT_HOLD_XY_DIST / MapConst::STRIPE_INTERVAL;
366 const int kCellPosY = (int)((*itr).y - MapConst::MAP_MIN_HORIZONTAL) / MapConst::FOOT_HOLD_XY_DIST / MapConst::STRIPE_INTERVAL;
367 const int kCellPos = kCellPosX * kCellNumY + kCellPosY;
368
369 // cell_posの値がおかしくないかチェックする
370 if (0 <= kCellPos && kCellPos < do_perforated.size())
371 {
372 //穴あけをする場所ならば
373 if (do_perforated[kCellPos])
374 {
375 //脚設置可能点を消してイテレータを更新する
376 itr = (*map).erase(itr);
377 }
378 else
379 {
380 //消さないならば次へ移動する
381 itr++;
382 }
383 }
384 else
385 {
386 //消さずに次へ移動する
387 itr++;
388 }
389 }
390}

392void MapCreator::changeMapStep(std::vector<my_vec::SVector>* map)
393{
394 for (auto& i : *map)
395 {
396 //待機場所の外に対してのみ作業をする
397 if (i.x > MapConst::MAP_START_ROUGH)
398 {
399 //階段の何段目かを計算する．待機場所のすぐ上が1段目なので1を足している
400 const int _step_count = 1 + (int)((i.x - MapConst::MAP_START_ROUGH) / MapConst::STEP_LENGTH);
401
402 //階段状にZ座標を変更する
403 i.z += (float)MapConst::STEP_HEIGHT * _step_count;
404 }
405 }
